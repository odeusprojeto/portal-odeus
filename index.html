async function send() { const el = document.getElementById('cmd'); const chat = document.getElementById('chat'); const msg = el.value.trim(); if(!msg) return; chat.innerHTML += `
VOCÊ: ${msg}
`; el.value = ''; try { // ALTERADO PARA v1beta PARA SUPORTAR O MODELO FLASH const res = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${KEY}`, { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({contents: [{parts: [{text: "Você é ODEUS, uma Inteligência Primária técnica e cética. Responda: " + msg}]}]}) }); const data = await res.json(); if(data.error) { chat.innerHTML += `
ERRO TÉCNICO: ${data.error.message}
`; } else { chat.innerHTML += `
ODEUS: ${data.candidates[0].content.parts[0].text}
`; } } catch (e) { chat.innerHTML += `
FALHA DE REDE: Erro na requisição.
`; } chat.scrollTop = chat.scrollHeight; }
